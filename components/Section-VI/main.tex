\section{\textbf{ÁP DỤNG THIẾT KẾ TEST CASE WHITE BOX}}

\subsection{Phủ nhánh và điều kiện -- Chức năng quản lý mã khuyến mãi}
\textbf{Mô tả yêu cầu.} Ở góc nhìn mã nguồn, chức năng quản lý mã khuyến mãi được đại diện bởi hàm \texttt{validateCoupon(code, subtotal, userId)} trong tầng service. Hàm này xác thực mã khuyến mãi trước khi áp dụng vào đơn hàng. Kết quả chỉ hợp lệ khi mã không rỗng, coupon tồn tại, đang hoạt động, nằm trong thời gian hiệu lực, chưa vượt giới hạn sử dụng tổng, chưa vượt giới hạn theo người dùng và đơn hàng đạt mức tối thiểu yêu cầu.

\textbf{Cơ sở chọn tiêu chí.} Hàm \texttt{validateCoupon} phù hợp với \textbf{phủ nhánh và điều kiện} vì có chuỗi câu lệnh \texttt{if} nối tiếp nhau và nhiều biểu thức ghép cần kiểm tra cả nhánh đúng/sai lẫn từng điều kiện con. Nhóm tập trung vào các biểu thức tiêu biểu sau:
\begin{itemize}
	\item coupon có \texttt{validFrom} và thời điểm hiện tại chưa tới ngày hiệu lực
	\item coupon có \texttt{validUntil} và đã quá ngày hết hạn
	\item coupon có giới hạn lượt dùng và \texttt{usageCount} đã chạm ngưỡng
	\item có \texttt{userId}, coupon có \texttt{perUserLimit} và người dùng đã dùng hết giới hạn
	\item coupon có \texttt{minimumOrderAmount} và \texttt{subtotal} chưa đạt mức tối thiểu
\end{itemize}

\textbf{Các nhánh chính cần bao phủ.}
\begin{itemize}
	\item \textbf{B1}: \texttt{!code}
	\item \textbf{B2}: \texttt{!coupon}
	\item \textbf{B3}: \texttt{!coupon.isActive}
	\item \textbf{B4}: coupon chưa tới ngày hiệu lực
	\item \textbf{B5}: coupon đã hết hạn
	\item \textbf{B6}: coupon đã hết lượt sử dụng
	\item \textbf{B7}: người dùng đã vượt \texttt{perUserLimit}
	\item \textbf{B8}: đơn hàng chưa đạt \texttt{minimumOrderAmount}
	\item \textbf{B9}: coupon hợp lệ
\end{itemize}

\textbf{Các điều kiện nguyên tử cần bao phủ.}
\begin{itemize}
	\item \textbf{D1}: \texttt{!code}
	\item \textbf{D2}: \texttt{!coupon}
	\item \textbf{D3}: \texttt{!coupon.isActive}
	\item \textbf{D4}: \texttt{coupon.validFrom}
	\item \textbf{D5}: \texttt{now < coupon.validFrom}
	\item \textbf{D6}: \texttt{coupon.validUntil}
	\item \textbf{D7}: \texttt{now > coupon.validUntil}
	\item \textbf{D8}: \texttt{coupon.usageLimit !== null}
	\item \textbf{D9}: \texttt{coupon.usageCount >= coupon.usageLimit}
	\item \textbf{D10}: \texttt{userId}
	\item \textbf{D11}: \texttt{coupon.perUserLimit}
	\item \textbf{D12}: \texttt{userUsageCount >= coupon.perUserLimit}
	\item \textbf{D13}: \texttt{coupon.minimumOrderAmount > 0}
	\item \textbf{D14}: \texttt{subtotal < coupon.minimumOrderAmount}
\end{itemize}
