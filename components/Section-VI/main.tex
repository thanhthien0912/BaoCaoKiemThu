\subsection{Phủ nhánh và điều kiện -- Chức năng quản lý mã khuyến mãi}
\textbf{Mô tả yêu cầu.} Ở góc nhìn mã nguồn, chức năng quản lý mã khuyến mãi được đại diện bởi hàm \texttt{validateCoupon(code, subtotal, userId)} trong tầng service. Hàm này xác thực mã khuyến mãi trước khi áp dụng vào đơn hàng. \textbf{Phân tích.} Dựa vào Hình 2, nhóm tiến hành áp dụng kiểm thử phủ nhánh và phủ điều kiện cho hàm \texttt{validateCoupon}. Viết theo đúng cấu trúc source, các điểm kiểm tra và điều kiện tương ứng trong hàm được xác định như sau:
\begin{itemize}
	\item \textbf{1}: \texttt{!code}
	\item \textbf{2}: \texttt{!coupon}
	\item \textbf{3}: \texttt{!coupon.isActive}
	\item \textbf{4.1}: \texttt{coupon.validFrom}
	\item \textbf{4.2}: \texttt{now < coupon.validFrom}
	\item \textbf{5.1}: \texttt{coupon.validUntil}
	\item \textbf{5.2}: \texttt{now > coupon.validUntil}
	\item \textbf{6.1}: \texttt{coupon.usageLimit !== null}
	\item \textbf{6.2}: \texttt{coupon.usageCount >= coupon.usageLimit}
	\item \textbf{7.1}: \texttt{userId}
	\item \textbf{7.2}: \texttt{coupon.perUserLimit}
	\item \textbf{8}: \texttt{userUsageCount >= coupon.perUserLimit}
	\item \textbf{9.1}: \texttt{coupon.minimumOrderAmount > 0}
	\item \textbf{9.2}: \texttt{subtotal < coupon.minimumOrderAmount}
	\item \textbf{10}: \texttt{return \{ valid: true, ... \}}
\end{itemize}

\subsection{Phủ nhánh -- Chức năng đăng nhập}
\textbf{Mô tả yêu cầu.} Ở góc nhìn mã nguồn, chức năng đăng nhập được đại diện bởi hàm \texttt{login(\{ email, password \})} trong \texttt{authService}. Hàm tiếp nhận thông tin đăng nhập, tìm người dùng theo email, kiểm tra mật khẩu và trả về thông tin người dùng kèm token nếu xác thực thành công.

\textbf{Hàm được chọn để phân tích.} Nhóm chọn hàm \texttt{login(\{ email, password \})} để thực hiện phân tích phủ nhánh cho chức năng đăng nhập.

\begin{figure}[H]
	\centering
	\includegraphics[width=\textwidth]{image/anhcodechuongwhitebox/codelogin.png}
	\caption{Mã nguồn hàm \texttt{login} dùng cho phân tích}
	\label{fig:login-code}
\end{figure}

\textbf{Cơ sở chọn tiêu chí.} Hàm \texttt{login} phù hợp với \textbf{phủ nhánh} vì cấu trúc xử lý ngắn, rõ ràng và gồm ba nhánh nghiệp vụ chính:
\begin{itemize}
	\item không tìm thấy người dùng theo email
	\item tìm thấy người dùng nhưng mật khẩu không khớp
	\item tìm thấy người dùng, mật khẩu đúng và đăng nhập thành công
\end{itemize}

\textbf{Các nhánh chính cần bao phủ.}
\begin{itemize}
	\item \textbf{B1}: \texttt{!user}
	\item \textbf{B2}: \texttt{!isMatch}
	\item \textbf{B3}: đăng nhập thành công, tạo token và trả về kết quả
\end{itemize}
