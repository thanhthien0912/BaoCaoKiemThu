\section{\textbf{ÁP DỤNG THIẾT KẾ TEST CASE BLACK BOX}}

\subsection{Bảng quyết định -- Chức năng quản lý mã khuyến mãi}
\textbf{Mô tả yêu cầu.} Dựa trên mã nguồn của hệ thống, chức năng quản lý khuyến mãi được triển khai tại trang quản trị \texttt{/admin/coupons}. Quản trị viên có thể tạo mới, chỉnh sửa, bật/tắt trạng thái và xóa mã khuyến mãi. Form nhập liệu gồm các trường chính: \texttt{code}, \texttt{name}, \texttt{description}, \texttt{discountType}, \texttt{discountValue}, \texttt{maximumDiscount}, \texttt{minimumOrderAmount}, \texttt{usageLimit}, \texttt{perUserLimit}, \texttt{validFrom}, \texttt{validUntil}, \texttt{isActive}. Ở tầng backend, hệ thống chỉ chấp nhận lưu khi mã và tên không rỗng, loại giảm giá hợp lệ, mã được chuẩn hóa sang chữ hoa và không bị trùng, giá trị phần trăm không vượt quá \texttt{100}, các giá trị số không âm, đồng thời bản ghi cần thao tác phải tồn tại. Riêng thao tác xóa chỉ được phép khi mã khuyến mãi chưa từng được sử dụng; nếu \texttt{usageCount > 0} thì hệ thống từ chối xóa và yêu cầu vô hiệu hóa thay thế.

\textbf{Cơ sở chọn kỹ thuật.} Chức năng này phù hợp với \textbf{bảng quyết định} vì kết quả xử lý phụ thuộc vào nhiều điều kiện kết hợp như thao tác thực hiện, dữ liệu bắt buộc, mã trùng, giá trị phần trăm, bản ghi tồn tại và trạng thái đã sử dụng. Bảng quyết định giúp rút gọn các tổ hợp chính thành các luật đại diện để sinh test case trong file Excel.

\textbf{Các điều kiện quyết định chính.}
\begin{itemize}
	\item \textbf{C1}: thao tác đang thực hiện là tạo/cập nhật hay xóa.
	\item \textbf{C2}: dữ liệu bắt buộc hợp lệ, tối thiểu gồm mã và tên không rỗng.
	\item \textbf{C3}: loại giảm giá hợp lệ theo tập \texttt{percentage}, \texttt{fixed\_amount}, \texttt{free\_shipping}.
	\item \textbf{C4}: nếu loại giảm là \texttt{percentage} thì \texttt{discountValue} không vượt quá \texttt{100}.
	\item \textbf{C5}: mã khuyến mãi không bị trùng với bản ghi khác sau khi chuẩn hóa sang chữ hoa.
	\item \textbf{C6}: bản ghi mục tiêu tồn tại khi cập nhật hoặc xóa.
	\item \textbf{C7}: mã khuyến mãi chưa có lượt sử dụng khi thực hiện xóa.
\end{itemize}

\textbf{Hướng sinh test case từ bảng quyết định.} Từ tám luật trên, nhóm sinh các test case đại diện như tạo mã hợp lệ, tạo mã trùng, tạo mã phần trăm vượt \texttt{100\%}, cập nhật coupon không tồn tại, xóa coupon chưa dùng và xóa coupon đã có \texttt{usageCount}. Test case chi tiết được lưu trong file Excel; báo cáo chỉ giữ phần phân tích.

\subsection{Phân vùng tương đương -- Chức năng đăng nhập}
\textbf{Mô tả yêu cầu:} chức năng đăng nhập tại \texttt{/login} yêu cầu xác thực qua email và password. Hệ thống chỉ xử lý khi dữ liệu không trống và email đúng định dạng. Để bảo mật, mật khẩu được ẩn mặc định và mọi lỗi xác thực đều trả về thông báo chung ``Email hoặc mật khẩu không đúng''. Ngoài ra, giao diện hỗ trợ bật/tắt hiển thị mật khẩu và tự động xóa thông báo lỗi ngay khi người dùng bắt đầu chỉnh sửa lại thông tin.

\textbf{Cơ sở chọn kỹ thuật:} chức năng này phù hợp với phân vùng tương đương do các trường đầu vào có tập hợp giá trị lớn. Kỹ thuật này giúp chia dữ liệu thành các nhóm đại diện hợp lệ và không hợp lệ, từ đó rút gọn số lượng test case mà vẫn đảm bảo bao phủ toàn bộ các nhánh xử lý của hệ thống, tránh việc kiểm thử lặp lại dư thừa.

\textbf{Các test case đăng nhập:}
\begin{itemize}
	\item TC01: Đăng nhập thành công với tài khoản hợp lệ
	\item TC02: Đăng nhập thất bại do sai Mật khẩu
	\item TC03: Đăng nhập thất bại do Email chưa đăng ký
	\item TC04: Kiểm tra thông báo lỗi khi để trống Email
	\item TC05: Kiểm tra thông báo lỗi khi để trống Mật khẩu
	\item TC06: Kiểm tra lỗi khi để trống cả Email và Mật khẩu
	\item TC07: Kiểm tra lỗi khi nhập Email sai định dạng
	\item TC08: Kiểm tra tính bảo mật (Mật khẩu ẩn mặc định)
	\item TC09: Kiểm tra chức năng Bật/Tắt hiển thị mật khẩu
	\item TC10: Kiểm tra tự động xóa thông báo lỗi khi sửa dữ liệu
\end{itemize}
